create stream ProductInventoryEnriched as 
  select ROWKEY, 
        pi.after->id as id, 
        pi.after->prd_id as productId, 
        i.after->id as inventoryId,
        pi.after->amount_of as amount_of, 
        i.after->stock as stock,
        (i.after->stock/pi.after->amount_of) as can_make
  from ProductInventory pi 
  inner join inventory i 
  WITHIN 7 days
  on i.after->id = pi.after->inv_id
  emit changes;

create table ProductInventoryEnriched_tbl as
  select id as id,
         latest_by_offset(productId) as productId,
         latest_by_offset(inventoryId) as inventoryId,
         latest_by_offset(amount_of) as amount_of,
         latest_by_offset(stock) as stock,
         latest_by_offset(can_make) as can_make
  from ProductInventoryEnriched
  group by id
  emit changes;
